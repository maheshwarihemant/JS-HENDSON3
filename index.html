Q2== Find the Fives.

        const Find_Five = (N) => 
{
  const numStr = N.toString();
  let five = 0;
  for (const ch of numStr){
    if (ch == "5"){
      five++;
      
    }
  }
  return five;
	 
};

const Prime_Check = (N) => 
{
  let factor=0;

    if(N%i===0){
      factor++;
    }
  }
  if(factor===2){

  }
  else{
    return 'NO'
  }
